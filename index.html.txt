        function selectRegName(name) {
            document.getElementById('regItemName').value = name;
            document.getElementById('regNameSuggestions').classList.add('hidden');
            const existingItem = inventoryCache.find(item => item['Item Name'] === name);
            if (existingItem && existingItem['Category']) document.getElementById('regCategory').value = existingItem['Category'];
        }

        function renderSuggestions(id, list, onSelect) {
            const container = document.getElementById(id);
            if (list.length === 0) { container.classList.add('hidden'); return; }
            container.innerHTML = list.map(b => `<div class="suggestion-item text-sm" onclick="this.parentElement.classList.add('hidden');">${b}</div>`).join('');
            container.querySelectorAll('.suggestion-item').forEach((item, index) => {
                item.addEventListener('click', () => { onSelect(list[index]); container.classList.add('hidden'); });
            });
            container.classList.remove('hidden');
        }

        function renderSearchSuggestions(list) {
            const container = document.getElementById('searchSuggestions');
            if (list.length === 0) { container.classList.add('hidden'); return; }
            container.innerHTML = list.map(item => `<div class="suggestion-item flex justify-between items-center text-sm gap-2" onclick="selectSearchItem('${item['Item Code']}')"><div class="flex flex-col"><span class="font-bold text-gray-800">${item['Item Name']}</span><span class="text-[10px] text-gray-500">${item['Branch'] || 'ไม่ระบุสาขา'}</span></div><span class="font-mono text-xs bg-indigo-50 text-indigo-600 px-1 rounded border border-indigo-100">${item['Item Code']}</span></div>`).join('');
            container.classList.remove('hidden');
        }

        function generateAutoCode() {

            let catPrefix = "EQ";

            else if (catVal.includes("กล้อง")) catPrefix = "CA";
            const branchText = document.getElementById('regBranch').value.trim();
            if (!branchText) { showToast("กรุณาเลือกสาขา", "red"); return; }
            const branchCode = branchText.substring(0, 2).padStart(2, '0');
            const dateStr = new Date().toISOString().slice(0,10).replace(/-/g, '');
            const mNo = (document.getElementById('regMachineNo').value || "1").padStart(2, '0');
            document.getElementById('regItemCode').value = `${catPrefix}${branchCode}${dateStr}${mNo}`;
        }

        async function loadAllDataForSearch(manual = false) {
            document.getElementById('dataStatus').innerText = "สถานะ: กำลังโหลด...";
            try {
                const res = await fetch(SCRIPT_URL);
                const json = await res.json();
                if(json.result === 'success') {
                    inventoryCache = json.data; 
                    document.getElementById('dataStatus').innerText = `สถานะ: ออนไลน์ (${inventoryCache.length} รายการ)`;
                    if(manual) showToast("อัปเดตข้อมูลแล้ว", "green");
                }
            } catch(e) { document.getElementById('dataStatus').innerText = "สถานะ: ออฟไลน์"; }
        }

        function showToast(msg, color="gray") {
            const t = document.getElementById('toast');
            t.innerText = msg;
            t.className = `fixed bottom-5 left-1/2 -translate-x-1/2 px-6 py-3 rounded-full shadow-lg text-sm z-50 text-white ${color === 'green' ? 'bg-green-600' : (color === 'red' ? 'bg-red-600' : 'bg-gray-800')}`;
            t.classList.remove('hidden');
            setTimeout(() => t.classList.add('hidden'), 2500);
        }

        document.getElementById('itemForm').onsubmit = async (e) => {
            e.preventDefault();
            const data = new URLSearchParams(new FormData(e.target));
            const btn = document.getElementById('submitBtn');
            btn.disabled = true;
            try {
                const res = await fetch(SCRIPT_URL, { method: 'POST', body: data });
                const json = await res.json();
                if(json.result === 'success') {
                    showToast("บันทึกสำเร็จ", "green");
                    const code = document.getElementById('regItemCode').value;
                    const name = document.getElementById('regItemName').value;
                    const c = document.getElementById('qrcode');
                    c.innerHTML = "";
                    new QRCode(c, { text: code, width: 150, height: 150 });
                    document.getElementById('qrCodeText').innerText = code;
                    document.getElementById('qrNameText').innerText = name;
                    document.getElementById('qrResult').classList.remove('hidden');
                    e.target.reset();
                    loadAllDataForSearch();
                }
            } catch(err) { showToast("เกิดข้อผิดพลาด", "red"); }
            btn.disabled = false;
        };

        function findItem(code) {
            document.getElementById('loadingMsg').classList.remove('hidden');
            document.getElementById('searchResultCard').classList.add('hidden');
            document.getElementById('notFoundMsg').classList.add('hidden');
            const item = inventoryCache.find(r => String(r['Item Code']).toLowerCase() === String(code).toLowerCase());
            setTimeout(() => {
                document.getElementById('loadingMsg').classList.add('hidden');
                if (item) {
                    document.getElementById('resName').innerText = item['Item Name'];
                    document.getElementById('resCode').innerText = "รหัส: " + item['Item Code'];
                    document.getElementById('resCode').dataset.code = item['Item Code'];
                    document.getElementById('resCategory').innerText = item['Category'];
                    document.getElementById('resBranch').innerText = item['Branch'];
                    document.getElementById('resMachineNo').innerText = item['MachineNo'] || '1';
                    document.getElementById('resSpec').value = item['Spec'] || '';
                    document.getElementById('resNote').value = item['Note'] || '';
                    const rc = document.getElementById('resQrcode');
                    rc.innerHTML = "";
                    new QRCode(rc, { text: item['Item Code'], width: 140, height: 140 });
                    document.getElementById('resQrName').innerText = item['Item Name'];
                    document.getElementById('resQrCode').innerText = item['Item Code'];
                    document.getElementById('searchResultCard').classList.remove('hidden');
                } else { document.getElementById('notFoundMsg').classList.remove('hidden'); }
            }, 300);
        }

        function startScanner() {
            const container = document.getElementById('scannerContainer');
            container.classList.remove('hidden');
            html5QrcodeScanner = new Html5Qrcode("reader");
            html5QrcodeScanner.start({ facingMode: "environment" }, { fps: 10, qrbox: 250 }, (text) => {
                stopScanner(); findItem(text); document.getElementById('searchInput').value = text;
            }).then(() => isScannerRunning = true);
        }

        function stopScanner() {
            if (html5QrcodeScanner && isScannerRunning) {
                html5QrcodeScanner.stop().then(() => { html5QrcodeScanner.clear(); document.getElementById('scannerContainer').classList.add('hidden'); isScannerRunning = false; });
            } else { document.getElementById('scannerContainer').classList.add('hidden'); }
        }

        function openPinModal(cb) { onPinSuccess = cb; document.getElementById('pinModal').classList.remove('hidden'); document.getElementById('pinInput').focus(); }

        async function performUpdate() {
            const code = document.getElementById('resCode').dataset.code;
            const data = new URLSearchParams({ action:'update', code:code, note:document.getElementById('resNote').value, spec:document.getElementById('resSpec').value });
            try {
                const res = await fetch(SCRIPT_URL, { method: 'POST', body: data });
                const json = await res.json();
                if(json.result === 'success') { showToast("อัปเดตข้อมูลแล้ว", "green"); loadAllDataForSearch(); }
            } catch(e) { showToast("อัปเดตไม่สำเร็จ", "red"); }
        }
